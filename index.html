function drawTimer() { ctx.clearRect(0, 0, canvas.width, canvas.height); const centerX = canvas.width / 2; const centerY = canvas.height / 2; const radius = canvas.width / 2 - 10; const startAngle = -Math.PI / 2; // ① 薄いグレーの円の輪郭を常時描画 ctx.beginPath(); ctx.arc(centerX, centerY, radius, 0, Math.PI * 2); ctx.strokeStyle = "#ddd"; // 薄いグレー ctx.lineWidth = 8; ctx.stroke(); // ② 経過時間の色を描画 const progress = elapsed / totalSeconds; const endAngle = startAngle + progress * Math.PI * 2; ctx.beginPath(); ctx.arc(centerX, centerY, radius, startAngle, endAngle); ctx.strokeStyle = isBreak ? "#4fc3f7" : "#ff7043"; // 休憩：水色、勉強：オレンジ ctx.lineWidth = 8; ctx.stroke(); }
